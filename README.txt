lnxterm - linux/docker/kubernetes terminal and file browser



-- feature
- no installation required
- zero configuration



-- download
wget "https://github.com/thread43/lnxterm/releases/download/v0.1/lnxterm_v0.1.tar.gz"
tar xzvf lnxterm_v0.1.tar.gz
cd lnxterm
./lnxterm



-- usage
./lnxterm
./lnxterm --help
./lnxterm --port=1234
./lnxterm --host="0.0.0.0"
./lnxterm --host="127.0.0.1"
./lnxterm --log="file"
./lnxterm --log="stdout"
./lnxterm --sqlite="lnxterm.db"
./lnxterm --mysql="root:123456@tcp(127.0.0.1:3306)/lnxterm"
